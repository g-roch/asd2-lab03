86 class BellmanFordSP : public ShortestPath<GraphType> {
87 
88 private:
89  typedef ShortestPath<GraphType> BASE;
90  typedef typename BASE::Edge Edge;
91  typedef typename BASE::Weight Weight;
92 
93  // Relachement de l'arc e
94  void relax(const Edge& e) {
95  int v = e.From(), w = e.To();
96  Weight distThruE = this->distanceTo[v]+e.Weight();
97 
98  if(this->distanceTo[w] > distThruE) {
99  this->distanceTo[w] = distThruE;
100  this->edgeTo[w] = e;
101  }
102  }
103 
104 public:
105 
106  // Constructeur a partir du graphe g et du sommet v a la source
107  // des plus courts chemins
108  BellmanFordSP(const GraphType& g, int v) {
109 
110  this->edgeTo.resize(g.V());
111  this->distanceTo.assign(g.V(),std::numeric_limits<Weight>::max());
112 
113  this->edgeTo[v] = Edge(v,v,0);
114  this->distanceTo[v] = 0;
115 
116  for(int i=0;i<g.V();++i)
117  g.forEachEdge([this](const Edge& e){
118  this->relax(e);
119  });
120  }
